## 二、PCF8574 IO扩展芯片

### 2.1 芯片简介
PCF8574是一个I2C接口的8位I/O扩展芯片，用于扩展Hi3861的GPIO口，可控制风扇、蜂鸣器、LED灯等外设。

### 2.2 硬件配置

```c
#define PCF8574_I2C_ADDR (0x43)  // I2C从机地址
#define PCF8574_I2C_ID 0        // I2C总线号
#define PCF8574_SPEED 100000    // 100KHz通信速率
```

**引脚配置**：
- GPIO_9: I2C_SCL (时钟线)
- GPIO_10: I2C_SDA (数据线)


```c:1:140:src/vendor/hqyj/fs_hi3861/common/bsp/src/hal_bsp_pcf8574.c
uint32_t PCF8574_Init(void)
{
    uint32_t result = 0;

    // gpio_9 复用为 I2C_SCL
    hi_io_set_pull(HI_IO_NAME_GPIO_9, HI_IO_PULL_UP);
    hi_io_set_func(HI_IO_NAME_GPIO_9, HI_IO_FUNC_GPIO_9_I2C0_SCL);
    // gpio_10 复用为 I2C_SDA
    hi_io_set_pull(HI_IO_NAME_GPIO_10, HI_IO_PULL_UP);
    hi_io_set_func(HI_IO_NAME_GPIO_10, HI_IO_FUNC_GPIO_10_I2C0_SDA);

    result = hi_i2c_init(PCF8574_I2C_ID, PCF8574_SPEED);
    if (result != HI_ERR_SUCCESS) {
        printf("I2C PCF8574 Init status is 0x%x!!!\r\n", result);
        return result;
    }
    printf("I2C PCF8574 Init is succeeded!!!\r\n");

    tmp_io.bit.p0 = 0; // 风扇关闭
    tmp_io.bit.p1 = 1; // 蜂鸣器关闭
    tmp_io.bit.p2 = 1; // LED关闭
    tmp_io.bit.p4 = 1; // 火焰传感器初始引脚
    tmp_io.bit.p5 = 0; // 可燃气体传感器
    tmp_io.bit.p7 = 1; // LED关闭

    PCF8574_Write(tmp_io.all); // v4.2版本开发板 关闭风扇，关闭蜂鸣器，关闭LED灯
    return HI_ERR_SUCCESS;
}
```

## 三、SHT20温湿度传感器

### 3.1 芯片简介
SHT20是一款数字温湿度传感器，通过I2C接口通信，提供高精度的温度和湿度测量。

### 3.2 硬件配置

```c
#define SHT20_I2C_ADDR (0x80)  // I2C从机地址
#define SHT20_I2C_IDX 0        // I2C总线号
#define SHT20_I2C_SPEED 100000 // 100KHz

// 寄存器地址
#define SHT20_HoldMaster_Temp_REG_ADDR 0xE3  // 主机模式温度寄存器
#define SHT20_HoldMaster_Humi_REG_ADDR 0xE5  // 主机模式湿度寄存器
#define SHT20_NoHoldMaster_Temp_REG_ADDR 0xF3 // 非主机模式温度寄存器
#define SHT20_NoHoldMaster_Humi_REG_ADDR 0xF5 // 非主机模式湿度寄存器
#define SHT20_SW_REG_ADDR 0xFE                // 软复位寄存器

// 时间参数
#define SHT20_READ_TEMP_DATA_TIME 85000   // 85ms
#define SHT20_READ_HUMI_DATA_TIME 50000   // 50ms
#define SHT20_RESET_TIME 100000            // 100ms
```

```c:68:114:src/vendor/hqyj/fs_hi3861/common/bsp/src/hal_bsp_sht20.c
uint32_t SHT20_ReadData(float *temp, float *humi)
{
    uint32_t result;
    uint8_t buffer[4] = {0};

    /* 发送检测温度命令 */
    result = SHT20_WiteByteData(SHT20_NoHoldMaster_Temp_REG_ADDR);
    if (result != HI_ERR_SUCCESS) {
        printf("I2C SHT20 status = 0x%x!!!\r\n", result);
        return result;
    }

    usleep(SHT20_READ_TEMP_DATA_TIME); /* datasheet: typ=66, max=85 */

    // 读数据
    result = SHT20_RecvData(buffer, READ_TEMP_DATA_NUM);
    if (result != HI_ERR_SUCCESS) {
        printf("I2C SHT20 status = 0x%x!!!\r\n", result);
        return result;
    }

    // 看芯片手册，手册中有转换公式的说明
    *temp = TEMP_VALUE_1 * (((((int)buffer[0]) << TEMP_LEFT_SHIFT_8) + buffer[1]) / TEMP_VALUE_2) - TEMP_VALUE_3;

    memset_s(buffer, sizeof(buffer), 0, sizeof(buffer));

    /* 发送检测湿度命令 */
    result = SHT20_WiteByteData(SHT20_NoHoldMaster_Humi_REG_ADDR);
    if (result != HI_ERR_SUCCESS) {
        printf("I2C SHT20 status = 0x%x!!!\r\n", result);
        return result;
    }

    usleep(SHT20_READ_HUMI_DATA_TIME); /* datasheet: typ=22, max=29 */

    // 读数据
    result = SHT20_RecvData(buffer, READ_HUMI_DATA_NUM);
    if (result != HI_ERR_SUCCESS) {
        printf("I2C SHT20 status = 0x%x!!!\r\n", result);
        return result;
    }

    // 看芯片手册，手册中有转换公式的说明
    *humi = HUMI_VALUE_1 * (((((int)buffer[0]) << HUMI_LEFT_SHIFT_8) + buffer[1]) / HUMI_VALUE_2) - HUMI_VALUE_3;

    return HI_ERR_SUCCESS;
}
```

**数据转换公式**：
```
温度(℃) = 175.72 × ((ST & 0xFFFC) / 65536.0) - 46.85
湿度(%) = 125 × ((SRH & 0xFFFC) / 65536.0) - 6
```

## 四、AW2013 RGB LED灯

```c:88:146:src/vendor/hqyj/fs_hi3861/common/bsp/src/hal_bsp_ap3216c.c
uint32_t AP3216C_ReadData(uint16_t *irData, uint16_t *alsData, uint16_t *psData)
{
    uint32_t result = 0;
    uint8_t data_H = 0, data_L = 0;

    // 读取IR传感器数据    10-bit
    result = AP3216C_ReadRegByteData(AP3216C_IR_L_ADDR, &data_L);
    if (result != HI_ERR_SUCCESS) {
        return result;
    }

    result = AP3216C_ReadRegByteData(AP3216C_IR_H_ADDR, &data_H);
    if (result != HI_ERR_SUCCESS) {
        return result;
    }

    if (data_L & 0x80) {
        // IR_OF为1, 则数据无效
        *irData = 0;
    } else {
        // 芯片数据手册中有数据转换的说明
        *irData = ((uint16_t)data_H << LEFT_SHIFT_2) | (data_L & 0x03);
    }

    // 读取ALS传感器数据    16-bit
    result = AP3216C_ReadRegByteData(AP3216C_ALS_L_ADDR, &data_L);
    if (result != HI_ERR_SUCCESS) {
        return result;
    }

    result = AP3216C_ReadRegByteData(AP3216C_ALS_H_ADDR, &data_H);
    if (result != HI_ERR_SUCCESS) {
        return result;
    }

    // 芯片数据手册中有数据转换的说明
    *alsData = ((uint16_t)data_H << LEFT_SHIFT_8) | (data_L);

    // 读取PS传感器数据    10-bit
    result = AP3216C_ReadRegByteData(AP3216C_PS_L_ADDR, &data_L);
    if (result != HI_ERR_SUCCESS) {
        return result;
    }

    result = AP3216C_ReadRegByteData(AP3216C_PS_H_ADDR, &data_H);
    if (result != HI_ERR_SUCCESS) {
        return result;
    }

    if (data_L & 0x40) {
        // IR_OF为1, 则数据无效
        *psData = 0;
    } else {
        // 芯片数据手册中有数据转换的说明
        *psData = ((uint16_t)(data_H & 0x3F) << LEFT_SHIFT_4) | (data_L & 0x0F);
    }

    return HI_ERR_SUCCESS;
}
```

```c
        AP3216C_ReadData(&ir, &als, &ps);
        
        // 显示状态
        char status[20];
        if (ir > 0) {
            sprintf(status, "Human:DETECTED");
            SSD1306_ShowStr(0, 2, (unsigned char *)status, TEXT_SIZE_16);
            
            // 报警：红灯闪烁 + 蜂鸣器
            AW2013_Control_RGB(255, 0, 0);
            set_buzzer(true);
            set_led(true);
        } else {
            sprintf(status, "Human: NONE   ");
            SSD1306_ShowStr(0, 2, (unsigned char *)status, TEXT_SIZE_16);
            
            // 正常：绿灯 + 关闭蜂鸣器
            AW2013_Control_RGB(0, 255, 0);
            set_buzzer(false);
            set_led(false);
        }
        
        sleep(1);
    }
}
```

## 八、外设总结

### 8.1 外设对比表

| 外设型号 | 功能 | I2C地址 | 通信协议 | 特点 |
|---------|------|----------|---------|------|
| PCF8574 | IO扩展 | 0x43 | I2C | 扩展8个IO口 |
| SHT20 | 温湿度 | 0x80 | I2C | 高精度温湿度 |
| AW2013 | RGB LED | 0x8A | I2C | 3通道PWM调光 |
| AP3216C | 三合一传感器 | 0x3C | I2C | 光照+接近+红外 |
| SSD1306 | OLED显示 | 0x78 | I2C | 128×64单色显示 |

### 8.2 共同特性

1. **通信总线**：所有外设均通过I2C总线通信
2. **GPIO引脚**：GPIO_9(I2C_SCL)和GPIO_10(I2C_SDA)
3. **通信速率**：100KHz
4. **初始化流程**：GPIO配置 → I2C初始化 → 寄存器配置

### 8.3 使用注意事项

1. **初始化顺序**：先初始化PCF8574，再初始化其他I2C外设
2. **时序要求**：部分外设需要延时等待数据稳定
3. **错误处理**：检查函数返回值，处理初始化失败情况
4. **资源管理**：多个I2C设备需要避免总线冲突
5. **低功耗设计**：使用OLED的休眠功能降低功耗

### 8.4 典型应用场景

| 应用场景 | 涉及外设 | 控制逻辑 |
|---------|----------|---------|
| 智能照明 | SHT20 + AW2013 + SSD1306 | 温度控制RGB灯亮度 |
| 环境监测 | SHT20 + AP3216C + SSD1306 | 显示温湿度/光照/红外 |
| 安防报警 | AP3216C + PCF8574 + AW2013 | 检测人体→LED+蜂鸣器 |
| 温湿度控制 | SHT20 + PCF8574 | 湿度控制风扇 |
| 状态指示 | PCF8574 + SSD1306 | 多个LED状态显示 |
| 数据显示 | SSD1306 | 传感器数据实时显示 |

```c
void system_init(void)
{
    // 1. 初始化I2C总线
```
